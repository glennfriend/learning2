Arrow Function


    var nums = [1,2,3];

    console.log(
        nums.map( function(v) {
            return v * 2;
        })
    );

    console.log(
        nums.map( (v) => v * 2)
    );

Default Parameter Values


    (function fn (x, y = 7, z = 42) {
        console.log([x, y, z]);
    })
    (1);

Rest Parameter


    (function fn (x, y, ...a) {
        console.log(a);
    })
    (1, 2, "hello", true, 7);
