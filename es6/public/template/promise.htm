setTimeout


    setTimeout(function(){
        console.log('touch 1')
        setTimeout(function(){
            console.log('touch 2')
        }, 1000)
    }, 1000)

setTimeout use Promise


    var wait1000 = () => new Promise((resolve, reject) => {setTimeout(resolve, 1000)})
    wait1000()
        .then(function() {
            console.log('touch 1')
            return wait1000()
        })
        .then(function() {
            console.log('touch 2')
        });
