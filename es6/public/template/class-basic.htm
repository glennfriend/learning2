Method Properties


    var obj = {
        name: null,
        setName (name) {
            this.name = name;
            return this;
        },
        getName (name) {
            return this.name;
        },
        show () {
            console.log(`hello ${this.getName()}`);
        }
    }
    obj.setName('ken').show();

Class Constructor


    class User {
        constructor () {
            this._name = null;
            this._age  = 0;
        }
        setName (name) {
            this._name = name;
            return this;
        }
        getName (name) {
            return this._name;
        }
        show () {
            console.log(`hello ${this.getName()} age is ${this._age}`);
        }
    }
    var user = new User;
    user.setName('ken').show();
