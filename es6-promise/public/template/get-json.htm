get success


    var promise = new Promise(function(resolve, reject){
        $.ajax({
            url: 'json/result-success.json',
            dataType: 'json',
            cache: false,
            success: resolve,
            error: reject
        });
    });

    promise
    .then(function(value){
        console.info(value);
    })
    .catch(function(error){
        console.warn(error);
    });
