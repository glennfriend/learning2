建立一個 success process


    var promise = new Promise(function(resolve, reject){
        resolve(200);
    });

    promise
    .then(function(value){
        console.info(value);
    })
    .catch(function(error){
        console.warn(error);
    });

建立一個 fail process


    var promise = new Promise(function(resolve, reject){
        reject(404);
    });

    promise
    .then(function(value){
        console.info(value);
    })
    .catch(function(error){
        console.warn(error);
    });
