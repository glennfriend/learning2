succes flow


    function taskA() {
        console.info("---- start ----");
        console.info("Task A");
    }
    function taskB() {
        console.info("Task B");
    }
    function onRejected(error) {
        console.error("Catch Error: A or B", error);
    }
    function finalTask() {
        console.info("Final Task");
    }

    Promise
        .resolve()
        .then(taskA)
        .then(taskB)
        .catch(onRejected)
        .then(finalTask);
